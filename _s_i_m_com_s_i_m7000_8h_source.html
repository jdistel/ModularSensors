28 // Status should be monitored on the STATUS pin, which is at a high level
29 // when the module has powered on and the firmware goes ready
30 // Time after end pulse until status pin becomes active (>4.5sec)
31 #define SIM7000_STATUS_LEVEL HIGH
32 #define SIM7000_STATUS_TIME_MS 5000L
33 
34 // R4 series are reset with a >10 SECOND low pulse on the RESET_N pin
35 #define SIM7000_RESET_LEVEL LOW
36 #define SIM7000_RESET_PULSE_MS 10000L
37 
38 // Module is switched on by a >1 second LOW pulse on the PWRKEY pin
39 // NOTE: Module is switched OFF by a >1.2 second LOW pulse on the PWRKEY pin,
40 // so by using a pulse of >1 but <1.2 s to wake the SIM7000 and using AT
41 // commands to put it to sleep, we should always be in the correct state
42 #define SIM7000_WAKE_LEVEL LOW
43 #define SIM7000_WAKE_PULSE_MS 1100
44 // Time after power on before "PWRKEY" can be used (guess - diagram isn't clear)
45 #define SIM7000_WARM_UP_TIME_MS 1000L
46 // Time after end pulse until serial port becomes active (>4.5sec)
47 #define SIM7000_ATRESPONSE_TIME_MS 4500
48 
49 // power down (gracefully) takes 1.8-6.9 sec
50 #define SIM7000_DISCONNECT_TIME_MS 7000L
