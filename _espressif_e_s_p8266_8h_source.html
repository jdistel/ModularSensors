30 // It is not possible to get status from the ESP8266 in deep sleep mode.
31 // During deep sleep the pin state is undefined
32 // For cases where a pin is defined for light sleep mode, the Espressif
33 // documentation states: since the system needs some time to wake up from
34 // light sleep, it is suggested that wait at least 5ms before sending next AT
35 // command. They don't say anything about the pin state though.
36 // The status level during light sleep is user selectable, but we set it low
37 // for wake and high for sleep. Of course, despite being able to configure
38 // light sleep mode for the module, it's not actually possible to purposefully
39 // enter light sleep via AT commands, so we are dependent on the module deciding
40 // it's been idle long enough and entering sleep on its own. That is a terrible
41 // system. Use a deep-sleep with reset if possible.
42 #define ESP8266_STATUS_LEVEL HIGH
43 #define ESP8266_STATUS_TIME_MS 350
44 
45 // Reset is very fast
46 #define ESP8266_RESET_LEVEL LOW
47 #define ESP8266_RESET_PULSE_MS 1
48 
49 // See notes above.. this is user configurable, but useless
50 #define ESP8266_WAKE_LEVEL LOW
51 #define ESP8266_WAKE_PULSE_MS 0
52 // Module turns on when power is applied regardless of pin states
53 #define ESP8266_WARM_UP_TIME_MS 0
54 // Time until system and digital pins are operational
55 #define ESP8266_ATRESPONSE_TIME_MS 350
56 
57 // power down ???
58 #define ESP8266_DISCONNECT_TIME_MS 500
