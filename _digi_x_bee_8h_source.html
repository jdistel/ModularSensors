25 // Status can either be measured directly with ON/SLP_N/DIO9 or indirectly with
26 // CTS_N/DIO9 The status level will depend on which is being used - the ON/SLP_N
27 // will be HIGH when the XBee is awake (ie, yes, I am not sleeping) but the
28 // CTS_N will be LOW when the board is away (ie, no it's not not clear to send).
29 // [double negatives much?] Time after turn on until the status pin is active -
30 // total WAG
31 #define XBEE_STATUS_TIME_MS 15
32 
33 // Reset is via a LOW pulse on the RESET_N pin
34 #define XBEE_RESET_LEVEL LOW
35 #define XBEE_RESET_PULSE_MS 5
36 
37 // Module is woken by holding DTR_N/SLP_RQ/DIO8 LOW (not pulsed)
38 #define XBEE_WAKE_LEVEL LOW
39 #define XBEE_WAKE_PULSE_MS 0
40 // Time after power-up before we can wake the model
41 #define XBEE_WARM_UP_TIME_MS 100
42 // Time we'll get back an AT response. Probably much less than this, except in
43 // bypass
44 #define XBEE_ATRESPONSE_TIME_MS 15000L
45 
46 // Power down time "can largely vary depending
47 // on the application / network settings and the concurrent module
48 // activities." Vint/status pin should be monitored and power not withdrawn
