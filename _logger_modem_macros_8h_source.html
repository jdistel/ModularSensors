28 // Check if the modem was awake using all possible means
29 // Don't want to accidently pulse an already on modem to off
30 /* NOTE: It's possible that the status pin is on, but the modem is
31  * actually mid-shutdown. In that case, we'll mistakenly skip
32  * re-waking it. This only applies to modules with a pulse wake (ie,
33  * non-zero wake time). For all modules that do pulse on, where
34  * possible I've selected a pulse time that is sufficient to wake but
35  * not quite long enough to put it to sleep and am using AT commands
36  * to sleep. This *should* keep everything lined up.*/
37 #define MS_IS_MODEM_AWAKE(specificModem) \
38  bool specificModem::isModemAwake(void) { \
39  if (_wakePulse_ms > 0 && _statusPin >= 0) { \
40  /* If there's a pulse wake up (ie, non-zero wake time) and there's \
41  * a status pin, use that to determine if the modem was awake \
42  * before setup began.*/ \
43  bool levelNow = digitalRead(_statusPin); \
44  MS_DBG(getModemName(), F("status pin"), _statusPin, F("level = "), \
45  levelNow ? F("HIGH") : F("LOW"), F("meaning"), \
46  getModemName(), F("should be"), \
47  levelNow == static_cast<int>(_statusLevel) ? F("on") \
48  : F("off")); \
49  return levelNow == static_cast<int>(_statusLevel); \
50  } else if (_wakePulse_ms == 0) { \
51  /* If the wake up is one where a pin is held (0 wake time) then \
52  * we're going to check the level of the held pin as the \
53  * indication of whether attempts were made to wake the modem \
54  * before entering the setup function*/ \
55  int8_t sleepRqBitNumber = \
56  log(digitalPinToBitMask(_modemSleepRqPin)) / log(2); \
57  int8_t currentRqPinState = bitRead( \
58  *portInputRegister(digitalPinToPort(_modemSleepRqPin)), \
59  sleepRqBitNumber); \
60  MS_DBG(F("Current state of sleep request pin"), _modemSleepRqPin, \
61  '=', currentRqPinState ? F("HIGH") : F("LOW"), \
62  F("meaning"), getModemName(), F("should be"), \
63  currentRqPinState == static_cast<int8_t>(_wakeLevel) \
64  ? F("on") \
65  : F("off")); \
66  return (currentRqPinState == static_cast<int8_t>(_wakeLevel)); \
67  } else if (_statusPin < 0) { \
68  /* If there's no status pin, but still a pulsed wake up, try \
69  * checking if the modem responds to AT commands*/ \
70  int8_t i = 5; \
71  bool res = false; \
72  while (i && !res) { \
73  gsmModem.sendAT(GF("")); \
74  res = gsmModem.waitResponse(100) == 1; \
75  if (res) break; \
76  delay(50); \
77  i--; \
78  } \
79  MS_DBG(F("Tested AT command and got"), \
80  res ? F("OK") : F("no response"), F("meaning"), \
81  getModemName(), \
82  res ? F("must be awake") : F("is probably asleep")); \
83  return res; \
84  } else { /*shouldn't get here*/ \
85  return true; \
86  } \
87  }

90 // The function to wake up the modem
91 #define MS_MODEM_WAKE(specificModem) \
92  bool specificModem::modemWake(void) { \
93  /* Power up */ \
94  if (_millisPowerOn == 0) { modemPowerUp(); } \
95  \
96  /* Set-up pin modes */ \
97  /* Because the modem calls wake BEFORE the first setup, we must set \
98  * the pin modes here */ \
99  setModemPinModes(); \
100  \
101  while (millis() - _millisPowerOn < _wakeDelayTime_ms) {} \
102  \
103  if (isModemAwake()) { \
104  MS_DBG(getModemName(), \
105  F("was already on! Will not run wake function.")); \
106  } else { \
107  /* Run the input wake function */ \
108  MS_DBG(F("Running wake function for"), getModemName()); \
109  if (!modemWakeFxn()) { \
110  MS_DBG(F("Wake function for"), getModemName(), \
111  F("did not run as expected!")); \
112  } \
113  } \
114  \
115  uint8_t resets = 0; \
116  bool success = false; \
117  while (!success && resets < 2) { \
118  /* Check that the modem is responding to AT commands */ \
119  MS_START_DEBUG_TIMER; \
120  MS_DBG(F("\nWaiting up to"), _max_atresponse_time_ms, F("ms for"), \
121  getModemName(), F("to respond to AT commands...")); \
122  success = gsmModem.testAT(_max_atresponse_time_ms + 500); \
123  if (success) { \
124  MS_DBG(F("... AT OK after"), MS_PRINT_DEBUG_TIMER, \
125  F("milliseconds!")); \
126  } else { \
127  MS_DBG(F("No response to AT commands!")); \
128  MS_DBG(F("Attempting a hard reset on the modem! "), \
129  resets + 1); \
130  if (!modemHardReset()) { \
131  /* Exit if we can't hard reset */ \
132  break; \
133  } else { \
134  resets++; \
135  } \
136  } \
137  } \
138  \
139  /* Clean any junk out of the modem buffer */ \
140  gsmModem.streamClear(); \
141  \
142  /* Re-run the modem init, or setup if necessary */ \
143  /* This will turn off echo, which often turns itself back on after \
144  * a reset/power loss */ \
145  /* This also checks the SIM card state */ \
146  if (!_hasBeenSetup) { \
147  /* If we run setup, take success value entirely from that*/ \
148  success = modemSetup(); \
149  } else { \
150  success &= gsmModem.init(); \
151  } \
152  gsmClient.init(&gsmModem); \
153  \
154  if (success) { \
155  modemLEDOn(); \
156  MS_DBG(getModemName(), F("should be awake and ready to go.")); \
157  } else { \
158  MS_DBG(getModemName(), F("failed to wake!")); \
159  } \
160  \
161  return success; \
162  }

237 // Get the time from NIST via TIME protocol (rfc868)
238 // This would be much more efficient if done over UDP, but I'm doing it
239 // over TCP because I don't have a UDP library for all the modems.
240 // NOTE: We eust ensure that we do not ping the daylight server more than once
241 // every 4 seconds. NIST clearly specifies here that this is a requirement for
242 // all software that accesses its servers:
243 // https://tf.nist.gov/tf-cgi/servers.cgi */
244 #define MS_MODEM_GET_NIST_TIME(specificModem) \
245  uint32_t specificModem::getNISTTime(void) { \
246  /* bail if not connected to the internet */ \
247  if (!isInternetAvailable()) { \
248  MS_DBG(F("No internet connection, cannot connect to NIST.")); \
249  return 0; \
250  } \
251  \
252  /* Try up to 12 times to get a timestamp from NIST */ \
253  for (uint8_t i = 0; i < 12; i++) { \
254  while (millis() < _lastNISTrequest + 4000) {} \
255  \
256  /* Make TCP connection */ \
257  MS_DBG(F("\nConnecting to NIST daytime Server")); \
258  bool connectionMade = gsmClient.connect("time.nist.gov", 37, 15); \
259  \
260  /* Wait up to 5 seconds for a response */ \
261  if (connectionMade) { \
262  uint32_t start = millis(); \
263  while (gsmClient && gsmClient.available() < 4 && \
264  millis() - start < 5000L) {} \
265  \
266  if (gsmClient.available() >= 4) { \
267  MS_DBG(F("NIST responded after"), millis() - start, \
268  F("ms")); \
269  byte response[4] = {0}; \
270  gsmClient.read(response, 4); \
271  if (gsmClient.connected()) gsmClient.stop(); \
272  return parseNISTBytes(response); \
273  } else { \
274  MS_DBG(F("NIST Time server did not respond!")); \
275  if (gsmClient.connected()) gsmClient.stop(); \
276  } \
277  } else { \
278  MS_DBG(F("Unable to open TCP to NIST!")); \
279  } \
280  } \
281  return 0; \
282  }
