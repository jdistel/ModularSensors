30 // Status can be monitored on the STATUS pin which is active HIGH
31 // Time after end pulse until status pin becomes active
32 // SIM800 - >3sec from start of 1s pulse
33 // SIM900 - >2.2sec from end of pulse
34 #define SIM800_STATUS_LEVEL HIGH
35 #define SIM800_STATUS_TIME_MS 3000
36 
37 // Reset with a >105ms low pulse on the RESET_N pin
38 #define SIM800_RESET_LEVEL LOW
39 #define SIM800_RESET_PULSE_MS 105
40 
41 // Module is switched on by a > 1 second LOW pulse on the PWR_ON pin
42 // Module is switched on by a 1-3 second LOW pulse on the PWR_ON pin
43 // Please monitor the status so on and off are correct!
44 #define SIM800_WAKE_LEVEL LOW
45 #define SIM800_WAKE_PULSE_MS 1100
46 // Time after power on before "PWRKEY" can be used - >0.4sec
47 #define SIM800_WARM_UP_TIME_MS 450
48 // Time after end pulse until serial port becomes active (>3sec from start of 1s
49 // pulse)
50 #define SIM800_ATRESPONSE_TIME_MS 3000
